272 void
273 syntreePrint(const syntree_t* self, syntree_nid root, FILE* out, int indent)
274 {
275  syntree_node_t* node = syntreeNodePtr(self, root);
276  fprintf(out, "%*s", indent*4, "");
277 
278  switch (node->tag)
279  {
280  case SYNTREE_TAG_Integer:
281  fprintf(out, "Integer %i\n", node->value.integer);
282  break;
283 
284  case SYNTREE_TAG_Float:
285  fprintf(out, "Float %g\n", node->value.real);
286  break;
287 
288  case SYNTREE_TAG_Boolean:
289  fputs((node->value.boolean ? "true\n" : "false\n"), out);
290  break;
291 
292  case SYNTREE_TAG_String:
293  fprintf(out, "\"%s\"\n", node->value.string);
294  break;
295 
296  case SYNTREE_TAG_LocVar:
297  case SYNTREE_TAG_GlobVar:
298  fprintf(out, "%s %s [pos=%i]\n",
299  node->tag == SYNTREE_TAG_LocVar ? "Local" : "Global",
300  nodeTypeName[node->type], node->value.variable);
301  break;
302 
303  case SYNTREE_TAG_Call:
304  fprintf(out, "Call %s [nid=%i] {\n", nodeTypeName[node->type],
305  node->value.container.last);
306  syntreePrint(self, node->value.container.first, out, indent+1);
307  fprintf(out, "%*s}\n", indent*4, "");
308  break;
309 
310  case SYNTREE_TAG_Program:
311  fprintf(out, "Program [globals=%u] {\n",
312  node->value.program.globals);
313  syntreePrint(self, node->value.program.body, out, indent+1);
314  fprintf(out, "%*s}\n", indent*4, "");
315  break;
316 
317  case SYNTREE_TAG_Function:
318  fprintf(out, "%s Function [locals=%u] {\n",
319  nodeTypeName[node->type], node->value.function.locals);
320  syntreePrint(self, node->value.function.body, out, indent+1);
321  fprintf(out, "%*s}\n", indent*4, "");
322  break;
323 
324  default:
325  if (node->type != SYNTREE_TYPE_Void)
326  fprintf(out, "%s ", nodeTypeName[node->type]);
327 
328  fprintf(out, "%s {\n", nodeTagName[node->tag]);
329 
330  for (root = node->value.container.first; root != 0;
331  root = syntreeNodePtr(self, root)->next)
332  {
333  syntreePrint(self, root, out, indent + 1);
334  }
335 
336  fprintf(out, "%*s}\n", indent*4, "");
337  }
338 }
