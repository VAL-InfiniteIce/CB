63 static int
64 locate(const dict_t* self, const char* key, locate_result_t* result)
65 {
66  enum { UNUSED, FORMER, MATCH } state = UNUSED;
67  hash_t hash = fnvHash(key);
68  const unsigned int mask = (1u << self->bits) - 1;
69  const unsigned int initial = hash & mask;
70 
71  /* berechne den initialen Index */
72  locate_result_t probe = { initial, &self->data[initial] };
73 
74  /* berechne die Schrittweite unter Einsatz weiterer Bits des Hashwertes;
75  * die Schrittweite entsteht aus der bitweisen Rotation des Hashwertes
76  * um die Anzahl der Bits in der Hashmap und ist immer ungerade (erstes
77  * Bit auf eins fixiert); der Grund dafür ist, dass die Größe der
78  * Hashmap eine Potenz von 2 und jede ungerade Zahl relativ prim dazu
79  * ist; mit dem euklidischen Algorithmus lässt sich zeigen, dass in
80  * diesem Fall alle Einträge in einer zufälligen Permutation besucht
81  * werden; die untere Schleife findet daher garantiert einen freien
82  * Platz, falls das Wörterbuch nicht zu 100% ausgelastet ist */
83  hash = (hash >> (self->bits - 1))
84  | (hash << (sizeof(hash)*CHAR_BIT - self->bits + 1))
85  | 1;
86 
87  /* diese Schleife läuft, bis wir den initialen Index erneut sehen */
88  do
89  {
90  /* teste, ob der aktuelle Eintrag schon in Benutzung war */
91  if (probe.p->key == NULL)
92  {
93  /* teste, ob wir bisher noch keinen freien Eintrag
94  * gefunden haben */
95  if (state == UNUSED)
96  *result = probe;
97 
98  /* Abbruch: der Eintrag war noch nie in Benutzung */
99  break;
100  }
101 
102  /* teste, ob der aktuelle Eintrag im Moment benutzt wird */
103  if (probe.p->val == NULL)
104  {
105  /* teste, ob es sich um das erste Mal handelt,
106  * dass wir einen freien Eintrag finden */
107  if (state == UNUSED)
108  {
109  *result = probe;
110  state = FORMER;
111  }
112  }
113  /* teste, ob dieser Eintrag der gesuchte ist */
114  else if (strcmp(probe.p->key, key) == 0)
115  {
116  *result = probe;
117  state = MATCH;
118  break;
119  }
120 
121  /* gehe zum nächsten Index */
122  probe.i = (probe.i + hash) & mask;
123  probe.p = &self->data[probe.i];
124  }
125  while (probe.i != initial);
126 
127  return state == MATCH;
128 }

134 static void
135 grow(dict_t* self)
136 {
137  dict_entry_t* data = self->data, *it, *end;
138  unsigned int cap = 1u << self->bits;
139 
140  /* alloziere ein doppelt so großes Array für die Einträge */
141  self->data = calloc(2*cap, sizeof(*self->data));
142 
143  /* teste auf Speicherfehler */
144  if (self->data == NULL)
145  {
146  fputs("out-of-memory error\n", stderr);
147  exit(-1);
148  }
149 
150  /* aktualisiere die Anzahl der Bits und Plätze */
151  ++self->bits;
152  self->left += cap;
153 
154  /* iteriere über das Originalarray und füge die Elemente neu ein */
155  for (it = data, end = it + cap; it < end; ++it)

160

164  }
