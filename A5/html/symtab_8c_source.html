125 int
126 symtabInsert(symtab_t* self, symtab_symbol_t* sym)
127 {
128  unsigned int i;
129 
130  /* berechne Kenngrößen */
131  sym->rec_prev = dictSet(&self->map, sym->name, sym);
132  sym->is_global = symtabIsGlobal(self);
133  sym->id = stackCount(self->decl);
134  sym->pos = sym->id;
135 
136  /* berechne die Position im Stack (für die Interpretation) */
137  if (!sym->is_global)
138  {
139  sym->pos -= self->block[0];
140 
141  if (sym->pos > self->maxpos)
142  self->maxpos = sym->pos;
143  }
144  else for (i = 0; i < stackCount(self->decl); ++i)
145  if (self->decl[i]->is_function)
146  --sym->pos;
147 
148  /* teste, ob schon eine Deklaration im gleichen Block vorliegt */
149  if (sym->rec_prev != NULL
150  && sym->rec_prev->id >= sym->id - stackTop(self->block))
151  {
152  /* stelle die Originaldefinition wieder her */
153  dictSet(&self->map, sym->name, sym->rec_prev);
154 
155  /* gib das Symbol frei */
156  symtabSymbolFree(sym);
157 
158  /* Rückgabe mit Fehlercode */
159  return -1;
160  }
161 
162  /* merke dir die Variable */
163  stackPush(self->decl) = sym;
164  ++stackTop(self->block);
165 
166  /* keine Fehler aufgetreten */
167  return 0;
168 }

276  nodeTypeName[sym->type], sym->name);
277
