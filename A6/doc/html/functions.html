  • bits : dict_t
  • block : symtab_t

  • boolean : syntree_node_t::syntree_node_value_u

- c -

  • cap : stack_hdr_t , syntree_t

- d -

  • data : dict_t
  • decl : symtab_t

- e -

  • eax : minako_vm_t
  • ebp : minako_vm_t
  • esp : minako_vm_t

- i -

  • i : locate_result_t
  • id : symtab_symbol_t
  • integer : syntree_node_t::syntree_node_value_u
  • is_function : symtab_symbol_t
  • is_global : symtab_symbol_t
  • is_param : symtab_symbol_t

- k -

  • key : dict_entry_t

- l -

  • left : dict_t
  • len : syntree_t , stack_hdr_t

- m -

  • map : symtab_t
  • maxpos : symtab_t

- n -

  • name : symtab_symbol_t
  • next : syntree_node_t
  • nodes : syntree_t

- p -

  • p : locate_result_t
  • par_next : symtab_symbol_t
  • pos : symtab_symbol_t

- r -

  • real : syntree_node_t::syntree_node_value_u
  • rec_prev : symtab_symbol_t

- s -

  • stack : minako_vm_t
  • string : syntree_node_t::syntree_node_value_u

- t -

  • tag : syntree_node_t
  • type : syntree_node_t , symtab_symbol_t , minako_value_t

- v -

  • val : dict_entry_t
  • variable : syntree_node_t::syntree_node_value_u
